nodejs-http

2017-09-08

今天遇到场景，导出年级报告（pdf），一个页面需要中间层（node）向底层请求50+数量接口提供数据，因为接口数量过多，数据请求完成还后需要生成pdf。不管怎么请求都会超时，用户端都会超时，平均一个报告至少需要1分钟
甚至更长。

技术方案：

  • Object.defineProperty set 避免深层回调
  • res.write(progress); 
let allUrlList = []//所有url接口
let allDataObject = {};
let itemUrl = Object.defineProperty({}, "num", {
        set:function(num){
            fetch(allUrlList[num],{...postJson},(data)=>{
            res.write(parseInt(num/allUrlList.length));
            allDataObject.merge(data);
            itemUrl.num = num+1;
        })
    }
});

疑问：也希望各位大神能帮忙解惑，同样的等待时间，为什么不断的，res.write(progress)就不会超时了？

另外产生一个问题：因为等待时间比较长，用户会主动关闭页面或者返回等终端请求，服务端的接口依然是在请求，对一些资源有不必要的浪费.
因此又有了新的方案，如下代码
原理是侦听close事件，这里要注意不是abort; 

const http = require("http");
const server = http.createServer(function(req,res){
        res.writeHead(200,{'Content-type':'text/html'})
        if (req.url=='\/favicon.ico') {
            res.end();
            return;
        }
        if(req.url=='/test'){
            setTimeout(function(){
                res.write("test");
                res.end()
            },15000)
        };
        let i = 0;
        let t = setInterval(function(){
            res.write("hello");
            console.log(i);
            if (i==10) {
                res.end();
                clearInterval(t);
            };
            i++
        },1000);
        res.on('finish', function () {
            console.log('server: response finish');
        });
        res.on('close', function () {
                console.log('server: response close');
                clearInterval(t);
            });
});



server.listen(8032);
server.on('listening',function(){
    console.log('ok, server is running');
});
